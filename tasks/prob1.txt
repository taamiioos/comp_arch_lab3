.data:
    sum NUMBER 0

.code:
    MOV rc, 0

loop:
    CMP rc, 1000
    JZ end_loop
    MOD r2, r1, 3
    CMP r2, 0
    JZ add_to_sum
    MOD r2, r1, 5
    CMP r2, 0
    JNZ next_iter

add_to_sum:
    ADD sum, r1

next_iter:
    INC rc
    JMP loop

end_loop:
    LOAD r1, sum
    OUTPUT r1
    HLT

.data:
    sum WORD 0

.code:
    ; инициализировать счетчик i значением 0
    MOV rc, 0

loop:
    ; проверить, достигнут ли конец диапазона чисел
    CMP rc, 1000
    JZ end_loop

    ; вычислить остаток от деления текущего числа на 3
    MOD r2, r1, 3
    ; проверить, равен ли остаток нулю
    CMP r2, 0
    JZ add_to_sum

    ; вычислить остаток от деления текущего числа на 5
    MOD r2, r1, 5
    ; проверить, равен ли остаток нулю
    CMP r2, 0
    JNZ next_iter

add_to_sum:
    ; прибавить текущее число к сумме
    ADD sum, r1

next_iter:
    ; увеличить счетчик i на 1
    INC rc
    JMP loop

end_loop:
    ; загрузить значение суммы в регистр r1
    LOAD r1, sum
    ; вывести сумму на экран
    OUTPUT r1

    ; остановить выполнение программы
    HLT

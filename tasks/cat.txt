.code
loop:
    ; считать символ из стандартного ввода в регистр r1
    INPUT r1

    ; проверить, достигнут ли конец файла
    CMP r1, 0xFF
    JZ end

    ; вывести символ из регистра r1 на экран
    OUTPUT r1

    ; вернуться к началу цикла чтения
    JMP loop

end:
    ; остановить выполнение программы
    HLT
